# Создаём новый проект
mkdir merion-git

# Переходим в созданную директорию
cd merion-git

# Выполняем инициализацию Git репозитория
git init

# Создаём новый файл
echo 'Bill Gates' > MICROSOFT.md

cat MICROSOFT.md - прочитать содержимое файла

# Готовим к коммиту
git add MICROSOFT.md

# Делаем коммит
git commit -m 'add MICROSOFT.md'

# Смотрим статус
git status

# Удаляем файл
rm MICROSOFT.md

git status

# С помощью команды git status можно посмотреть статус репозитория

# git diff - разница между коммитами,просматривают перед коммитом

# git log - выводит все коммиты

- git branch (имя ветки) - создает новую ветку коммитов

- git checkout (имя ветки) - переход между ветками

- git merge (имя ветки) - слияние веток

- HEAD - это указывает в какой ветке находишься и на каком коммите

- Tag - постоянная метка для конкретного коммита

- git switch (имя ветки) - команда для перехода с веток

- git switch -c (имя ветки) - создает новую ветку и переходит в нее

- git rm MICROSOFT.md - удаление файла в текущем состоянии (не из истории), затем нужно сделать коммит, чтоб удалить вообще

- git restore index.md - это откат изменении сделанное в файле, до того, как сделать add .

- git restore --staged (имя файла) - это откат изменения, когда подготовил файл к коммиту после add . 

- Коммит при помощи Vim (текстовый редактор),запускается: git commit -> выбирается команда I (INSERT) -> вводится коммент -> ESC -> :wq

- git commit --amend -m "изменение текста последнего комита"

- git reset --soft HEAD~1 - это удаление последнего коммита, но до уровня stage

- git reset HEAD~1 - это удаление последнего коммита, до уровня unstaged

- git reset --hard HEAD~1 - это удаление из диска и все его изменения

- rm (имя файла) - удаление с unstaged

- git branch --d (имя ветки) - удаление ветки

- git switch (id коммита) -> изменение в редакторе -> сделать коммит и сохранить в новой ветки -> слить их вместе - это dettached commit

- git .gitkeep - это файл который создается в директории, которая пустая, но нужно чтоб сохранялась и отображалась в коммите

- git stash - перемещение изменении в файле, в другую ветку

- git stash list - список стеш

- git stash apply - перенесение в данную ветку

- git stash push -m "коммент для стеша"

- git stash apply "цифра" - перемещение по стешам

- git stash pop "цифра" - удаление стеша

- git stash clean - удаление всего стеша

- git reflog - список наших действии, там можно найти Id удаленного коммита и восстановить: git reset --hard Id
для восстановление ветки, так же находим его последний коммит и создаем новую ветку туда и закидываем

- git merge --squash (имя ветки) - переносит только изменения с одной ветки в другую, но историю коммитов не переносит, потом сам сделаешь один коммит где все эти изменении сделанные в той ветке

- git rebase - это команда для вливание коммитов с А на В, но с переносом ее поверх последнего коммита ветки В (последовательно А-В-С-"другой коммит"), а при помощи merge пойдет разветвление

- git cherry-pick (id) - это копирование определенного коммита с одной ветки в другую

- shh-keygen - создание shh ключа в windows

- git remote add origin (ssh репозитория с GitHub) - подключение GitHub репозитория

- git push -u origin (имя ветки) - отправляешь ветку в репозиторию, дальнейшие отправки коммитов: - git push, при отправке другой ветки прописывается - git push origin (имя ветки)

- git fetch - посмотреть ветки созданные в удаленном доступе (GitHub)

- git branch --track (имя ветки созданное удаленно) origin (имя ветки) - он только создаёт локальную ветку, отслеживающую удалённую, для стягивании - git checkout (имя ветки), - git pull

- git push --set-upstream origin (имя ветки) - создает удаленную ветку в GitHub, с помощью него при pull/push не нужно вводить имя ветки

- git clone (shh ключ с удаленного репозитория) - он скачает репозитории из GitHub одну ветку default, при - git checkout (имя другой ветки) стянет его тоже

- git branch -D --remote origin/имя ветки - удаление ветки, затем нужно сделать: - git push origin --delete (имя ветки)
